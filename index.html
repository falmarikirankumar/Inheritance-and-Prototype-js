3. Add a method to calculate sum of all elements in Array in array's prototype, use that method to calculate sum for multiple arrays. 

  const array = {
    arrTotal: function (arr) {
      let sum = 0;
      for (let i = 0; i < arr.length; i++) {
        sum += arr[i];
      }
      console.log(sum);
    },
  };

  Array.__proto__ = array;
  arr1 = [1, 2, 3, 4];
  arr2 = [11, 12, 13, 14];
  Array.arrTotal(arr1);
  Array.arrTotal(arr2);
